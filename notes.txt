# Generate two new llvm file from the original one

dungnguyen@tsunami:~/mcsema/llvm-3.5/lib/Transforms/test$ llvm-dis inline_asm.bc -o=inline_asm.ll

# Build and generate a CFG in new format
dungnguyen@tsunami:~/mcsema/llvm-3.5/lib/Transforms$ clang++-3.5 -std=c++11 objdump2CFG.cpp -o objdump2CFG -I../../include
dungnguyen@tsunami:~/mcsema/llvm-3.5/lib/Transforms/test$ clang-3.5 -c inline_asm.ll -O2

# Use the new lifter to generate lifted function in llvm
dungnguyen@tsunami:~/squirrel/squirrel-x86_64$ cp ~/mcsema/llvm-3.5/lib/Transforms/test/inline_asm.cfg .
dungnguyen@tsunami:~/squirrel/squirrel-x86_64$ ./lift inline_asm.cfg; cp ./lift_result.ll ~/mcsema/llvm-3.5/lib/Transforms/test; subl lift_result.ll
dungnguyen@tsunami:~/squirrel/squirrel-x86_64$ opt-3.5 -S -O2 lift_result.ll > lifted_inline_asm.ll; cp ./lifted_inline_asm.ll ~/mcsema/llvm-3.5/lib/Transforms/test; subl lifted_inline_asm.ll
dungnguyen@tsunami:~/mcsema/llvm-3.5/lib/Transforms/test$ llvm-as lifted_inline_asm.ll -o=lifted_inline_asm.bc

# Link two bitcode files

dungnguyen@tsunami:~/mcsema/llvm-3.5/lib/Transforms/test$ llvm-dis merge.bc -o=merge.ll; subl merge.ll
